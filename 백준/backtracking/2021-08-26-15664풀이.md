import java.util.*;
public class Main {
	static int n,m;
	static int input[];
	static Set<String> res;
	static ArrayList<String> rst;
	public static void main(String[] args){
		
		Scanner sc = new Scanner(System.in);
		n = sc.nextInt();
		m = sc.nextInt();
		input = new int[n];
		
		for (int i = 0; i < n; i++) {
			input[i] = sc.nextInt();
		}
		res = new HashSet<String>();
		rst = new ArrayList<String>();
		
		Arrays.sort(input);
		comb(0,0);
		for(String result:rst) {
			System.out.print(result);
		}
	}
	public static void comb(int idx, int flag) {
		if(idx == m) {
			String str = new String();
			for (int i = 0; i < n; i++) {
				if((flag&1<<i) != 0) {
					str += input[i] + " ";
				}
			}str += "\n";
			if(!res.contains(str)) {
				res.add(str);
				rst.add(str);
			}
			return;
		}
		for (int i = idx; i < n; i++) {
			if((flag&1<<i) != 0) continue;
				comb(idx+1,flag|1<<i);
		}
	}
}
