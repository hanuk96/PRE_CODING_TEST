import java.util.*;
public class Main
{	
	static int[]d;
	public static void main(String[] args) {
	
		Scanner sc = new Scanner(System.in);
		int tc = sc.nextInt();
		d = new int [tc + 1];
		d[0] = 0;
		d[1] = 1;
		d[2] = 2;
		
		System.out.println(tile(d,tc));
	}
	
	static int tile(int []arr, int target) {
		int result = 0;
		for (int i = 3; i <= target; i++) {
			d[i] = d[i - 1] + d[i - 2];
			d[i] %= 10007;
		}
		return result = d[target];
	}
}
