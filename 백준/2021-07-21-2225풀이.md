import java.util.*;

public class Main {
	public static void main(String args[]){
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int k = sc.nextInt();
		int num = n;
		
		int[][] dp = new int[201][201];//n,n-1,n-2 ... 1을 2개 뽑을 확률 초기화
		
		for(int i = 1; i <= k ;i++) {//i개를 뽑아 0을 만드는거는 00000000... 뿐이라 1로 초기화
			dp[i][0] = 1;
		}
		
		for(int i = 1; i <= k ; i++) {//뽑을 갯수
			for(int j = 1; j <= n ; j++){//0 ~ 정해준n까지
				dp[i][j] = (dp[i-1][j] + dp[i][j-1]) % 1000000000;
				num--;
			}
		}
		System.out.println(dp[k][n]);
	}
}
