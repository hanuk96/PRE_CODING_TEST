import java.util.*;
public class Main {
	static int n,c;
	static int[]input;
	static int[]result;
	public static void main(String[] args){
		Scanner sc = new Scanner(System.in);
		n = sc.nextInt();
		c = sc.nextInt();
		
		input = new int[n];
		result = new int[n];
		
		for (int i = 0; i < input.length; i++) {
			input[i] = sc.nextInt();
		}
		Arrays.sort(input);
		comb(0);
	}
	public static void comb(int idx) {
		if(idx == c) {
			for (int i = 0; i < c - 1; i++) {
				if(result[i] > result[i+1]) return;
			}
			
			for (int i = 0; i < c; i++) {
				System.out.print(result[i] + " ");
			}System.out.println();
			
			return;
		}
		for (int i = 0; i < n; i++) {
			result[idx] = input[i];
			comb(idx+1);
		}
	}
}
