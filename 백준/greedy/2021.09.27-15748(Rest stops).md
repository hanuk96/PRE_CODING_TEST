import java.util.*;
class rest implements Comparable<rest>{
	long dis,cost;
	rest(long dis,long cost){
		this.dis = dis;
		this.cost = cost;
	}
	public int compareTo(rest o) {
		if(this.cost == o.cost) {
			if(this.dis < o.dis)
				return -1;
			else
				return 1;
		}
		else if(this.cost > o.cost)
			return -1;
		else
			return 1;
	}
}
public class Main {
	public static void main(String[] args){
		Scanner sc = new Scanner(System.in);
		long l = sc.nextLong();
		long n = sc.nextLong();
		long rF = sc.nextLong();
		long rB = sc.nextLong();
		
		ArrayList<rest> pq = new ArrayList<>();
		for (int i = 0; i < n; i++) {
			int x = sc.nextInt();//쉬는 시간
			int c = sc.nextInt();//풀의 가치
			pq.add(new rest(x,c));
		}
		Collections.sort(pq);
		
		long result = 0;
		long now = 0;//현재 거리
		
		for(rest it : pq) {
			rest cur = it;
			if(cur.dis < now) continue;
			result += (rF - rB) * (cur.dis - now) * cur.cost;
			now = cur.dis;
		}
		System.out.println(result);
	}
}
