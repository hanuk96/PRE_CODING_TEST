## 14500 테트리미노

---
### [ 문제설명 ]

- 맵의 크기와 맵이 입력으로 주어진다. 맵은 정수형 숫자로 이루어져 있다.

- 도형 5개가 주어진다. 5개의 도형은 회전시킬수도 대칭시킬수도 있다. 

- 5개의 도형을 맵에 넣었을 때 값의 최댓값을 구하는 것이다.


---

```cpp
#include <iostream> 
#include <string>
#include <algorithm>
#define MAX 100

using namespace std;

int N, M, answer,cnt;
int map[MAX][MAX] = { 0, };
bool visited[MAX][MAX];
int dx[5] = {0,1,-1,0};
int dy[5] = {1,0,0,-1};

int DFS(int x, int y, int count) {
	int sum = 0;
	
	if (count == 4) return map[y][x];//마지막 depth
	
	sum += map[y][x];

	for (int i = 0; i < 4 ; i++){
		int nx = x + dx[i];// 다음 x
		int ny = y + dy[i];// 다음 y

		if (nx >= 0 && ny >= 0 && nx < M && ny < N && !visited[ny][nx]){
			visited[ny][nx] = true;
			sum = max(sum,map[y][x]+DFS(nx,ny,count + 1));
			visited[ny][nx] = false;
		}
	}
	return sum;
}
int EXEC(int x, int y) {
	int sum = 0;
	//ㅗ ㅡ센터
	if (y >= 1 && x >= 1 && x < M - 1)
		sum = max(sum, map[y][x - 1] + map[y][x] + map[y - 1][x] + map[y][x + 1]);
	//ㅏ ㅣ센터
	if (y >= 1 && y < N - 1 && x < M - 1)
		sum = max(sum, map[y - 1][x] + map[y][x] + map[y][x + 1] + map[y + 1][x]);
	//ㅜ ㅡ센터
	if (y >= 0 && y < N - 1 && x < M - 1)
		sum = max(sum, map[y][x - 1] + map[y][x] + map[y + 1][x] + map[y][x + 1]);
	//ㅓ ㅣ센터
	if (y >= 1 && y < N - 1 && x >= 1)
		sum = max(sum, map[y - 1][x] + map[y][x] + map[y][x - 1] + map[y + 1][x]);

	return sum;
}
int main() {
	cin >> N >> M;

	for (int i = 0; i < N; i++)
		for (int j = 0; j < M; j++)
			cin >> map[i][j];
	
	answer = 0;
	cnt = 0;

	for (int i = 0; i < N; i++) {
		for (int j = 0; j < M; j++) {
			visited[i][j] = true;
			answer = max(answer,DFS(i,j,1));
			answer = max(answer,EXEC(i,j));//-,| 세개일때 가운데다가 적립
			visited[i][j] = false;
		}
	}

	cout << answer <<endl;
}
```
