기존의 최장 길이만 구하는 방식과는 다르게, 그 증가하는 수열을 출력해야하는 문제였다
모든 경우의 수를 구해서, 그중 가장 긴 값을 출력하게끔 해주었다.
시간초과가 날거같았지만 다행이도 나지않았다 어려웠던 문제

---
```java
import java.util.*;
public class Main {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int arr[] = new int[n];
		int dp[] = new int[n];
		for (int i = 0; i < n; i++) {
			arr[i] = sc.nextInt();
		}
		
		ArrayList<Integer> result[];
		result = new ArrayList[n];
		
		for (int i = 0; i < n; i++) {
			result[i] = new ArrayList<Integer>();
		}
		
		int max = 0;
		int resultIdx = -1;
		for (int i = 0; i < n; i++) {
			dp[i] = 1;
			result[i].add(arr[i]);
			for (int j = 0; j < n; j++) {
				if(arr[i] > arr[j]) {
					dp[i] = Math.max(dp[j] + 1, dp[i]);
					if(dp[i] == dp[j] + 1) {
						result[i].clear();
						for(int k = 0; k < result[j].size();k++) {
							result[i].add(result[j].get(k));
						}
						result[i].add(arr[i]);
					}
				}
			}
			max = Math.max(max, dp[i]);
			if(max == dp[i]) {
				resultIdx = i;
			}
		}
		System.out.println(max);
		for (int i = 0; i < result[resultIdx].size(); i++) {
			System.out.print(result[resultIdx].get(i) + " ");
		}
	}
}
