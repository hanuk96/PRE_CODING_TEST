### 2615 오목
1. 6개 이상 같은수면 안됨
2. 방향은 오른쪽, 오른쪽 위, 아래, 오른쪽 아래 방향만 탐색해주면 됨 -> 왼쪽 맨 위, 아래 가장자리 값을 출력해주기 위함임
3. 19x19기 때문에 완탐을 돌려도 160000 정도라 가능!

---
```java
import java.util.*;
public class Main {
	static int[][]ba;
	static int dx[] = {0,1,1,-1};//오른쪽, 오른쪽 아래, 아래, 오른쪽 위
	static int dy[] = {1,1,0,1};
	public static void main(String[] args){
		Scanner sc = new Scanner(System.in);
		
		ba = new int[19][19];
		for (int i = 0; i < 19; i++) {
			for (int j = 0; j < 19; j++) {
				ba[i][j] = sc.nextInt();
			}
		}
		
		for (int i = 0; i < 19; i++) {
			for (int j = 0; j < 19; j++) {
				if(ba[i][j] != 0) {
					if(check(i,j,ba[i][j]) == true) {
						System.out.println((ba[i][j]) + "\n" + (i+1) + " " + (j+1));
						return;
					}
				}
			}
		}
		System.out.println(0);
	}
	private static boolean check(int x,int y, int index) {
		for (int i = 0; i < 4; i++) {
			int nx = x + dx[i];
			int ny = y + dy[i];
			
			int count = 1;
			while(true) {
				if(nx<19 && ny<19 && ny>=0 && nx>=0) {//범위내에 있으면
					if(ba[nx][ny] == index) {
						count++;
						nx = nx + dx[i]; ny = ny + dy[i];
					}
					else break;
				}
				else break;
			}
			if(count == 5) {
				if(x-dx[i]>=0 && y-dy[i]>=0 && x-dx[i]<19 && y-dy[i]<19) {
					if(ba[x-dx[i]][y-dy[i]] == index)
						return false;
				}
				return true;
			}
		}
		return false;
	}
}
